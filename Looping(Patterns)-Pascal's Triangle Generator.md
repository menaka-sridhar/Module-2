def generate_pascals_triangle(n):
    for i in range(n):
        num = 1
        for j in range(i + 1):
            print(num, end=" ")
            num = num * (i - j) // (j + 1)
        print()
rows = int(input("Enter the number of rows: "))
print("Pascal's Triangle:")
generate_pascals_triangle(rows)
